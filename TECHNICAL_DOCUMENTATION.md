```python
def normalize_name(name):
    """
    –ù–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è –∏–º–µ–Ω–∏ –¥–ª—è —Å–æ–ø–æ—Å—Ç–∞–≤–ª–µ–Ω–∏—è
    
    Args:
        name: str - –ò—Å—Ö–æ–¥–Ω–æ–µ –∏–º—è
    
    Returns:
        str - –ù–æ—Ä–º–∞–ª–∏–∑–æ–≤–∞–Ω–Ω–æ–µ –∏–º—è
    """
    if pd.isna(name) or name == '':
        return ''
    
    # 1. –ü—Ä–∏–≤–µ–¥–µ–Ω–∏–µ –∫ —Å—Ç—Ä–æ–∫–µ –∏ —É–¥–∞–ª–µ–Ω–∏–µ –ª–∏—à–Ω–∏—Ö –ø—Ä–æ–±–µ–ª–æ–≤
    normalized = str(name).strip()
    
    # 2. –ü—Ä–∏–≤–µ–¥–µ–Ω–∏–µ –∫ –∑–∞–≥–ª–∞–≤–Ω—ã–º –±—É–∫–≤–∞–º
    normalized = normalized.upper()
    
    # 3. –£–¥–∞–ª–µ–Ω–∏–µ –º–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω—ã—Ö –ø—Ä–æ–±–µ–ª–æ–≤
    normalized = re.sub(r'\s+', ' ', normalized)
    
    # 4. –¢—Ä–∞–Ω—Å–ª–∏—Ç–µ—Ä–∞—Ü–∏—è –∫–∏—Ä–∏–ª–ª–∏—Ü—ã –≤ –ª–∞—Ç–∏–Ω–∏—Ü—É
    cyrillic_to_latin = {
        '–ê': 'A', '–ë': 'B', '–í': 'V', '–ì': 'G', '–î': 'D', '–ï': 'E', '–Å': 'E',
        '–ñ': 'ZH', '–ó': 'Z', '–ò': 'I', '–ô': 'Y', '–ö': 'K', '–õ': 'L', '–ú': 'M',
        '–ù': 'N', '–û': 'O', '–ü': 'P', '–†': 'R', '–°': 'S', '–¢': 'T', '–£': 'U',
        '–§': 'F', '–•': 'KH', '–¶': 'TS', '–ß': 'CH', '–®': 'SH', '–©': 'SHCH',
        '–™': '', '–´': 'Y', '–¨': '', '–≠': 'E', '–Æ': 'YU', '–Ø': 'YA'
    }
    
    for cyr, lat in cyrillic_to_latin.items():
        normalized = normalized.replace(cyr, lat)
    
    # 5. –£–¥–∞–ª–µ–Ω–∏–µ —Å–ø–µ—Ü–∏–∞–ª—å–Ω—ã—Ö —Å–∏–º–≤–æ–ª–æ–≤
    normalized = re.sub(r'[^\w\s]', '', normalized)
    
    return normalized

def find_name_match(target_name, df, name_column):
    """
    –ü–æ–∏—Å–∫ —Å–æ–≤–ø–∞–¥–µ–Ω–∏—è –∏–º–µ–Ω–∏ –≤ DataFrame
    
    Args:
        target_name: str - –¶–µ–ª–µ–≤–æ–µ –∏–º—è –¥–ª—è –ø–æ–∏—Å–∫–∞
        df: pd.DataFrame - DataFrame –¥–ª—è –ø–æ–∏—Å–∫–∞
        name_column: str - –ù–∞–∑–≤–∞–Ω–∏–µ –∫–æ–ª–æ–Ω–∫–∏ —Å –∏–º–µ–Ω–∞–º–∏
    
    Returns:
        pd.DataFrame - –°—Ç—Ä–æ–∫–∏ —Å —Å–æ–≤–ø–∞–¥–µ–Ω–∏—è–º–∏
    """
    if df.empty or name_column not in df.columns:
        return pd.DataFrame()
    
    # 1. –¢–æ—á–Ω–æ–µ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ
    exact_match = df[df[name_column].str.upper() == target_name.upper()]
    if not exact_match.empty:
        return exact_match
    
    # 2. –°–æ–≤–ø–∞–¥–µ–Ω–∏–µ –ø–æ—Å–ª–µ –Ω–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏–∏
    df_normalized = df.copy()
    df_normalized['normalized'] = df_normalized[name_column].apply(normalize_name)
    target_normalized = normalize_name(target_name)
    
    normalized_match = df_normalized[df_normalized['normalized'] == target_normalized]
    if not normalized_match.empty:
        return normalized_match.drop('normalized', axis=1)
    
    # 3. –ß–∞—Å—Ç–∏—á–Ω–æ–µ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ (—Å–æ–¥–µ—Ä–∂–∏—Ç —Ü–µ–ª–µ–≤–æ–µ –∏–º—è)
    partial_match = df[df[name_column].str.contains(target_name, case=False, na=False)]
    if not partial_match.empty:
        return partial_match
    
    # 4. –û–±—Ä–∞—Ç–Ω–æ–µ —á–∞—Å—Ç–∏—á–Ω–æ–µ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ (—Ü–µ–ª–µ–≤–æ–µ –∏–º—è —Å–æ–¥–µ—Ä–∂–∏—Ç)
    reverse_match = df[df[name_column].apply(lambda x: target_name.lower() in str(x).lower() if pd.notna(x) else False)]
    if not reverse_match.empty:
        return reverse_match
    
    return pd.DataFrame()
```

```python
def calculate_monthly_total(result_df, idx, work_hours, employee_errors):
    """
    –†–∞—Å—á—ë—Ç –∏—Ç–æ–≥–∞ –∑–∞ —Ç–µ–∫—É—â–∏–π –º–µ—Å—è—Ü –ø–æ —Ñ–æ—Ä–º—É–ª–µ:
    –ò—Ç–æ–≥ = N - –í—Ä–µ–º—è –±–µ–∑ –æ—Ç–≥—É–ª–æ–≤ - –û—Å—Ç–∞—Ç–æ–∫ –∑–∞ –ø–æ–∑–∞–ø—Ä–æ—à–ª—ã–π –º–µ—Å—è—Ü + –û—Ç–≥—É–ª—ã
    
    Args:
        result_df: pd.DataFrame - DataFrame —Å –¥–∞–Ω–Ω—ã–º–∏
        idx: int - –ò–Ω–¥–µ–∫—Å —Å—Ç—Ä–æ–∫–∏ —Å–æ—Ç—Ä—É–¥–Ω–∏–∫–∞
        work_hours: float - N (—Ä–∞–±–æ—á–∏–µ —á–∞—Å—ã –≤ –º–µ—Å—è—Ü)
        employee_errors: list - –°–ø–∏—Å–æ–∫ –æ—à–∏–±–æ–∫ –¥–ª—è —Å–æ—Ç—Ä—É–¥–Ω–∏–∫–∞
    """
    try:
        # –ü–æ–ª—É—á–µ–Ω–∏–µ –∑–Ω–∞—á–µ–Ω–∏–π –∏–∑ DataFrame
        remainder = result_df.at[idx, '–û—Å—Ç–∞—Ç–æ–∫ –∑–∞ –ø–æ–∑–∞–ø—Ä–æ—à–ª—ã–π –º–µ—Å—è—Ü'] or 0
        time_worked = result_df.at[idx, '–í—Ä–µ–º—è –±–µ–∑ –æ—Ç–≥—É–ª–æ–≤'] or 0
        time_off = result_df.at[idx, '–û—Ç–≥—É–ª—ã'] or 0
        
        # –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ –≤ —á–∏—Å–ª–∞ —Å –æ–±—Ä–∞–±–æ—Ç–∫–æ–π NaN
        remainder = float(remainder) if pd.notna(remainder) else 0
        time_worked = float(time_worked) if pd.notna(time_worked) else 0
        time_off = float(time_off) if pd.notna(time_off) else 0
        
        # –ü—Ä–∏–º–µ–Ω–µ–Ω–∏–µ —Ñ–æ—Ä–º—É–ª—ã
        total = work_hours - time_worked - remainder + time_off
        
        # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞
        result_df.at[idx, '–ò—Ç–æ–≥ –∑–∞ —Ç–µ–∫—É—â–∏–π –º–µ—Å—è—Ü'] = total
        
        # –õ–æ–≥–∏—Ä–æ–≤–∞–Ω–∏–µ –¥–ª—è –æ—Ç–ª–∞–¥–∫–∏
        print(f"–†–∞—Å—á—ë—Ç –¥–ª—è {result_df.at[idx, '–§–ò–û']}: {work_hours} - {time_worked} - {remainder} + {time_off} = {total}")
        
    except Exception as e:
        employee_errors.append(f"–û—à–∏–±–∫–∞ —Ä–∞—Å—á—ë—Ç–∞ –∏—Ç–æ–≥–∞: {str(e)}")
```
